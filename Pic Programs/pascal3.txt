#include <stdio.h>
#include <stdlib.h>
/* run this program using the console pauser or add your own getch, system("pause") or input loop */
int main() 
{
	int i,j,k,n,z;
	n=5;
	z=1;
	for(i=1;i<=n;i++)//row i=4
	{
		for(j=1;j<=n-i;j++)//1<=1
		{
			printf(" ");
		}
		for(k=1;k<=z;k++)
		{
			printf("*");	
		}
		z=z+2;//1+2=3+2=5+2=7
		printf("\n");
	}
	for(i=1;i<=4;i++)
	{
			printf("   ***\n");
	}
	
}
